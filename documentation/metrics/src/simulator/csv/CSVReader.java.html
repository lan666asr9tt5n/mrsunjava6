3   import java.io.BufferedReader;
4   import java.io.File;
5   import java.io.FileReader;
6   import java.io.IOException;
7   import java.util.StringTokenizer;
8   import java.util.Vector;
9   
10  public class CSVReader {
11  
12      private File input;
13      private int seperator;
14      private int textSeperator;
15  
16      public CSVReader(File input,int seperator,int textSeperator) {
17          this.input = input;
18          this.seperator = seperator;
19          this.textSeperator = textSeperator;
20      }
21  
22      public CSVResult getCSVResult() throws IOException {
23          CSVResult result = null;
24          BufferedReader in = null;
25  
26          try {
27              String line = "";
28              Vector<CSVRow> rows = new Vector<CSVRow>();
29  
30              in = new BufferedReader(
31                      new FileReader(this.input)
32              );
33  
34              while((line=in.readLine()) != null) {
35                  StringTokenizer st = new StringTokenizer(line,String.valueOf((char)this.seperator));
36                  Vector<String> content = new Vector<String>();
37  
38                  while(st.hasMoreElements()) {
39                      content.add(st.nextToken());
40                  }
41                  rows.add(new CSVRow(content,this.seperator,this.textSeperator));
42              }
43  
44              result = new CSVResult(rows);
45          } finally {
46              if(in != null) {
47                  in.close();
48              }
49          }
50  
51          return result;
52      }
