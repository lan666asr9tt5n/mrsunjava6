9   
10  import javax.swing.JButton;

13

22  public class GraphicalInput extends AbstractInput {
23  
24      private GraphicalOutput graphicalOutput;    
25      
26      public GraphicalInput(AbstractController controller2, GraphicalOutput gO) {
27          super(controller2); 
28          this.graphicalOutput = gO;
29      }   
30      @Override
31      public void workLoop() {
32          JFrame frame = new JFrame("Simulation control");
33          frame.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
34          frame.setLayout(new GridLayout(0, 1, 2, 8));
35  
36          JButton addHighButton = new JButton("Add high glycemic food");
37          frame.add(addHighButton);
38          addHighButton.addActionListener(new ActionListener() {
39              public void actionPerformed(ActionEvent aE) {
40                  controller.addFood(AbstractController.FoodType.HIGH);
41              }
42          });
43  
44          JButton addMedButton = new JButton("Add med glycemic food");
45          frame.add(addMedButton);
46          addMedButton.addActionListener(new ActionListener() {
47              public void actionPerformed(ActionEvent aE) {
48                  controller.addFood(AbstractController.FoodType.MED);                
49              }
50          });
51  
52          JButton addLowButton = new JButton("Add low glycemic food");
53          frame.add(addLowButton);
54          addLowButton.addActionListener(new ActionListener() {
55              public void actionPerformed(ActionEvent aE) {
56                  controller.addFood(AbstractController.FoodType.LOW);                
57              }
58          });
59          
60          JButton addRapidInsulinButton = new JButton("Add rapid acting insulin");
61          frame.add(addRapidInsulinButton);
62          addRapidInsulinButton.addActionListener(new ActionListener() {
63              public void actionPerformed(ActionEvent aE) {
64                  controller.addInsulin(AbstractController.InsulinType.RAPID);                
65              }
66          });
67          
68          JButton addShortInsulinButton = new JButton("Add short acting insulin");
69          frame.add(addShortInsulinButton);
70          addShortInsulinButton.addActionListener(new ActionListener() {
71              public void actionPerformed(ActionEvent aE) {
72                  controller.addInsulin(AbstractController.InsulinType.SHORT);                
73              }
74          });
75          
76          JButton addLongInsulinButton = new JButton("Add long acting insulin");
77          frame.add(addLongInsulinButton);
78          addLongInsulinButton.addActionListener(new ActionListener() {
79              public void actionPerformed(ActionEvent aE) {
80                  controller.addInsulin(AbstractController.InsulinType.LONG);             
81              }
82          });
83  
84          JToggleButton toggleChartButton = new JToggleButton("Hide chart");
85          frame.add(toggleChartButton);
86          toggleChartButton.addActionListener(new ActionListener() {
87              public void actionPerformed(ActionEvent aE) {
88                  graphicalOutput.toggleVisibility();
89              }
90          });
91  
92          JToggleButton togglePauseButton = new JToggleButton("Pause");
93          frame.add(togglePauseButton);
94          togglePauseButton.addActionListener(new ActionListener() {
95              public void actionPerformed(ActionEvent aE) {
96                  controller.setPause();
97              }
98          });
99  
100         frame.pack();
101         frame.setVisible(true);
102     }
