12  public abstract class AbstractFood {
13  
14      // Amount of food which is eaten (currently just 1) in serving units.
15      protected int ammount;
16      // Carbohydrates in gram for 1 amount of food.
17      protected float carbonHydrates;
18      // The time when the food was eaten.
19      protected double time;
20  
21      public AbstractFood(int ammount, double time, float carbs) {
22          this.ammount = ammount;
23          this.time = time;
24          this.carbonHydrates = carbs;
25          System.out.println(this);
26      }
27  
28      /**
29       * Calculate the amount of glucose in mmol/l released to the blood at a given time.
30       */
31      abstract double calculateGlucose(double time);
32  
33      @Override
34      public String toString() {
35          return this.getClass().getName() + ":" + Model.getDateFromDouble(time)
36                  + " - " + this.ammount + " _ " + this.carbonHydrates;
37      }
38      
39      /**
40       * Get carbohydrates in grams.
41       * @return
42       */
43      public float getCarboHydrates(){
44          return this.carbonHydrates * this.ammount;
45      }
