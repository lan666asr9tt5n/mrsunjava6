6   /**
7    * Simplyfied assumption of a high glycemic food is a softdrink.
8    * Serving unit assumed is 250ml.
9    * @author rc
10   *
11   */
12  public class HighGlycemicFood extends AbstractFood {
13  
14      /**
15       * @param ammount
16       * @param time
17       */
18      public HighGlycemicFood(int ammount, double time) {
19          super(ammount, time, 26);
20      }
21  
22      /* (non-Javadoc)
23       * @see simulator.modules.food.AbstractFood#calculateGlucose(int)
24       */
25      @Override
26      double calculateGlucose(double time) {
27          double localTime = time - this.time;
28          return Math.sin(((double)localTime)/this.ammount);
29      }
